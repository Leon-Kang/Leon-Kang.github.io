void strChange(char s[]) { 
    for (int i = 0; s[i] != '\0'; i++) {
        if (s[0] > 'a' && s[0] < 'z') {
            s[0] -= ('a' - 'A');
        }
        if (s[i] == '_') {
            s[i + 1] -= ('a' - 'A');
            for (int j = i; s[j] != '\0'; j++) {
                s[j] = s[j + 1];
            }
        }   
    }
}
